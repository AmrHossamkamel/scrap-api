💡 ملاحظة: عند الإرسال المجمع، سيتم ملء Variables الموجودة هنا فقط بناءً على البيانات المستخرجة. أضف أسماء المفاتيح التي تريد استخدامها من JSON كـ Variables فارغة مثل {"field_name": ""}

أمثلة Query شائعة:

📋 جلب جميع البيانات

query {
  users {
    id
    name
    email
    created_at
  }
}

🔍 البحث مع شروط

query GetUserByEmail($email: String!) {
  users(where: {email: {_eq: $email}}) {
    id
    name
    email
  }
}

➕ إضافة بيانات جديدة

mutation InsertUser($name: String!, $email: String!) {
  insert_users_one(object: {name: $name, email: $email}) {
    id
    name
    email
  }
}

✏️ تحديث البيانات

mutation UpdateUser($id: Int!, $name: String!) {
  update_users_by_pk(pk_columns: {id: $id}, _set: {name: $name}) {
    id
    name
    email
  }
}

🗑️ حذف البيانات

mutation DeleteUser($id: Int!) {
  delete_users_by_pk(id: $id) {
    id
    name
  }
}
